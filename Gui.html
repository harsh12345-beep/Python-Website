from tkinter import *
root = Tk()
root.geometry("260x200")
root.title("Simple Calculator")
root.configure(background="gray")
root.minsize(250, 150)
root.maxsize(250, 150)
x = ""
def press(num):
global x
x = x + str(num)
a.set(x)
def equal():
try:
global x
total = str(eval(x))
a.set(total)
x = ""
except:
a.set("error")
x = ""
def clear():
global x
x =""
a.set("")
a = StringVar()
b = Entry(root, textvar=a)
b.grid(columnspan=4, ipadx=70)
Button1 = Button(root, text="1", bg='black', fg='green', height=1, width=5, command=lambda: press(1))
